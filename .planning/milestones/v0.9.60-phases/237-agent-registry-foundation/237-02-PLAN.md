```js
async function readPersistedAgentRegistry() {
  try {
    var stored = await chrome.storage.session.get([FSB_AGENT_REGISTRY_STORAGE_KEY]);
    var payload = stored ? stored[FSB_AGENT_REGISTRY_STORAGE_KEY] : null;
    if (!payload || typeof payload !== 'object') return null;
    if (payload.v !== FSB_AGENT_REGISTRY_PAYLOAD_VERSION) return null; // future-migration hook
    return payload;
  } catch (e) {
    return null; // do not throw; SW boot must not be poisoned
  }
}

async function writePersistedAgentRegistry(records) {
  try {
    var nextRecords = records && typeof records === 'object' ? records : {};
    if (Object.keys(nextRecords).length === 0) {
      await chrome.storage.session.remove(FSB_AGENT_REGISTRY_STORAGE_KEY);
      return;
    }
    var payload = {};
    payload[FSB_AGENT_REGISTRY_STORAGE_KEY] = {
      v: FSB_AGENT_REGISTRY_PAYLOAD_VERSION,
      records: nextRecords
    };
    await chrome.storage.session.set(payload);
  } catch (e) {
    // best-effort; do not throw
  }
}
```

```js
AgentRegistry.prototype.hydrate = async function() {
  if (this._hydrated) return;
  var self = this;
  return withRegistryLock(async function() {
    if (self._hydrated) return; // double-check after lock acquisition

    var payload = await readPersistedAgentRegistry();
    var records = payload && payload.records ? payload.records : {};

    // Step 1: rebuild Maps from persisted records
    Object.keys(records).forEach(function(agentId) {
      var record = records[agentId];
      self._agents.set(agentId, {
        agentId: record.agentId || agentId,
        createdAt: record.createdAt || Date.now(),
        tabIds: Array.isArray(record.tabIds) ? record.tabIds.slice() : []
      });
      var tabs = Array.isArray(record.tabIds) ? record.tabIds : [];
      self._tabsByAgent.set(agentId, new Set(tabs));
      tabs.forEach(function(tabId) { self._tabOwners.set(tabId, agentId); });
    });

    // Step 2: reconcile against live tabs; drop ghost records
    var liveTabIds = new Set();
    try {
      var liveTabs = await chrome.tabs.query({});
      liveTabs.forEach(function(t) { if (typeof t.id === 'number') liveTabIds.add(t.id); });
    } catch (e) {
      // If query fails, do not drop anything (be conservative)
      self._hydrated = true;
      return;
    }

    var reapedThisWake = [];
    var tabOwnerSnapshot = Array.from(self._tabOwners.entries());
    tabOwnerSnapshot.forEach(function(entry) {
      var tabId = entry[0], agentId = entry[1];
      if (!liveTabIds.has(tabId)) {
        reapedThisWake.push({ agentId: agentId, tabId: tabId, reason: 'tab_not_found' });
        self._tabOwners.delete(tabId);
        var setRef = self._tabsByAgent.get(agentId);
        if (setRef) {
          setRef.delete(tabId);
          // Note: 237 keeps the agent record even with empty tab set.
          // Phase 241 will add "release agent on pool.size === 0".
          // BUT for ghost-record reaping on hydrate specifically: if ALL tabs
          // for an agent are ghosts, drop the agent record too (this is the
          // "ghost agent" case, not the "agent legitimately has no tabs" case).
          if (setRef.size === 0) {
            self._tabsByAgent.delete(agentId);
            self._agents.delete(agentId);
          }
        }
      }
    });

    // Step 3: emit one diagnostic event per reaped record (rate-limited)
    reapedThisWake.forEach(function(reap) {
      emitAgentReapedEvent(reap.agentId, reap.tabId, reap.reason);
    });

    // Step 4: write reconciled snapshot back if anything changed
    if (reapedThisWake.length > 0) {
      await self._persist();
    }

    self._hydrated = true;
  });
};
```

```js
function emitAgentReapedEvent(agentId, tabId, reason) {
  var event = {
    type: 'agent:reaped',
    agentId: agentId,
    tabId: tabId,
    reason: reason,
    timestamp: Date.now(),
    agentIdShort: formatAgentIdForDisplay(agentId)
  };
  // Lazy-reference: helpers may not be loaded yet (Pitfall 5).
  if (typeof globalThis !== 'undefined' && typeof globalThis.rateLimitedWarn === 'function') {
    globalThis.rateLimitedWarn(
      FSB_AGENT_LOG_PREFIX,                                         // 'AGT' -> [FSB AGT]
      FSB_AGENT_REAP_RATE_LIMIT_CATEGORY_BASE + '-' + reason,        // 'agent-reaped-tab_not_found'
      'agent reaped',
      { agentIdShort: event.agentIdShort, tabId: tabId, reason: reason }
    );
  }
  return event;
}
```

```js
function createChromeTabsMock(initialTabs) {
  var tabs = (initialTabs || []).map(function(t) { return Object.assign({}, t); });
  var listeners = { onRemoved: [] };
  return {
    async query(_filter) { return tabs.slice(); },
    async get(tabId) {
      var found = tabs.find(function(t) { return t.id === tabId; });
      if (!found) {
        var err = new Error('No tab with id: ' + tabId);
        // Mimic chrome runtime: caller checks chrome.runtime.lastError OR catches throw
        throw err;
      }
      return found;
    },
    onRemoved: {
      addListener(fn) { listeners.onRemoved.push(fn); },
      _emit(tabId) { listeners.onRemoved.forEach(function(fn) { fn(tabId); }); }
    },
    _setTabs(newTabs) { tabs = newTabs.slice(); },
    _addTab(t) { tabs.push(Object.assign({}, t)); },
    _removeTab(tabId) { tabs = tabs.filter(function(t) { return t.id !== tabId; }); }
  };
}

function setupChromeMock(opts) {

    runtime: { id: 'phase-237-test', lastError: null },

    tabs: tabs
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| chrome.storage.session -> in-memory Maps | Storage payload is parsed and used to populate authoritative Maps. A tampered or corrupted payload could inject ghost records or invalid agentIds. |
| chrome.tabs.query -> liveTabIds Set | Live-tab list is the truth source for reconciliation. If query throws or returns stale data, ghosts could be retained or live records could be dropped. |
| LOG-04 ring buffer -> diagnostic export | Diagnostic events flow into a buffer that may be exported by user via the Sync tab. Sensitive data in `redactedContext` would surface there. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-237-02-01 | Tampering | Storage payload corruption (write interrupted by SW eviction) leaves invalid envelope; hydrate could read a partial record and crash | mitigate | `readPersistedAgentRegistry` returns `null` on `payload.v !== 1` or non-object payload; hydrate falls back to empty Maps. Each record's `tabIds` defensively coerced to Array via `Array.isArray(record.tabIds) ? ... : []`. Test 6 (version mismatch) and test 8 (corrupt envelope) cover this. |
| T-237-02-02 | Tampering | Ghost-record reaping during `hydrate` while a concurrent `registerAgent` runs causes interleaved Map mutations and inconsistent state | mitigate | `hydrate` body wrapped in `withRegistryLock` (matches RESEARCH.md Pattern 4). Test 12 stress-tests this with a Promise.all([hydrate, registerAgent]) and asserts both ran cleanly with correct final state. |
| T-237-02-03 | Information Disclosure | Diagnostic event leaks task content or full agent_id via `redactedContext` to the LOG-04 ring buffer | mitigate | Per CONTEXT.md D-03, the redactedContext payload contains ONLY: `{ agentIdShort, tabId, reason }`. `agentIdShort` is the 6-char prefix (D-02), tabId is a numeric, reason is an enum string.
