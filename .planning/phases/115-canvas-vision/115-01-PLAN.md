The script MUST:

1. Guard against double-installation: check `if (window.__canvasCallLog) return;`

2. Initialize globals:
```javascript
window.__canvasCallLog = [];
window.__canvasCallCount = 0;
window.__canvasInterceptorInstalled = true;
```

3. Define MAX_LOG = 5000

4. Wrap these CanvasRenderingContext2D.prototype methods by saving the original and replacing with a logging wrapper:
   - Drawing: fillRect, strokeRect, clearRect
   - Text: fillText, strokeText
   - Path: beginPath, closePath, moveTo, lineTo, bezierCurveTo, quadraticCurveTo
   - Shapes: arc, arcTo, ellipse, rect
   - Rendering: fill, stroke
   - Images: drawImage
   - Transform: setTransform, save, restore

5. Each wrapper MUST:
   - Increment `window.__canvasCallCount`
   - If `window.__canvasCallLog.length < MAX_LOG`, push a log entry
   - Log entry format: `{ m: methodName, a: serializedArgs }`
   - For args serialization: map each arg -- if typeof === 'object', store '[object]', otherwise store the raw value
   - For style-capturing methods (fillRect, strokeRect, fillText, strokeText, fill, stroke), also capture:
     - `fs: this.fillStyle`
     - `ss: this.strokeStyle`
     - `f: this.font`
     - `lw: this.lineWidth`
   - Call and return the original method with `original.apply(this, arguments)`

6. Add a `getCanvasScene()` function on `window` that summarizes the log into structured data:
```javascript
window.getCanvasScene = function() {
  const log = window.__canvasCallLog || [];
  if (log.length === 0) return { texts: [], rects: [], paths: [], totalCalls: 0 };

  const texts = [];
  const rects = [];
  const paths = [];
  let currentPath = null;

  for (const entry of log) {
    switch (entry.m) {
      case 'fillText':
      case 'strokeText':
        texts.push({
          text: String(entry.a[0]),
          x: Math.round(entry.a[1]),
          y: Math.round(entry.a[2]),
          font: entry.f || '',
          color: entry.m === 'fillText' ? (entry.fs || '') : (entry.ss || ''),
          method: entry.m
        });
        break;
      case 'fillRect':
      case 'strokeRect':
        rects.push({
          type: entry.m,
          x: Math.round(entry.a[0]),
          y: Math.round(entry.a[1]),
          w: Math.round(entry.a[2]),
          h: Math.round(entry.a[3]),
          fill: entry.fs || '',
          stroke: entry.ss || '',
          lineWidth: entry.lw || 1
        });
        break;
      case 'beginPath':
        currentPath = { points: [], fill: null, stroke: null, lineWidth: 1 };
        break;
      case 'moveTo':
        if (currentPath) currentPath.points.push({ op: 'M', x: Math.round(entry.a[0]), y: Math.round(entry.a[1]) });
        break;
      case 'lineTo':
        if (currentPath) currentPath.points.push({ op: 'L', x: Math.round(entry.a[0]), y: Math.round(entry.a[1]) });
        break;
      case 'arc':
        if (currentPath) currentPath.points.push({ op: 'arc', cx: Math.round(entry.a[0]), cy: Math.round(entry.a[1]), r: Math.round(entry.a[2]) });
        break;
      case 'bezierCurveTo':
        if (currentPath) currentPath.points.push({ op: 'C', x: Math.round(entry.a[4]), y: Math.round(entry.a[5]) });
        break;
      case 'quadraticCurveTo':
        if (currentPath) currentPath.points.push({ op: 'Q', x: Math.round(entry.a[2]), y: Math.round(entry.a[3]) });
        break;
      case 'ellipse':
        if (currentPath) currentPath.points.push({ op: 'ellipse', cx: Math.round(entry.a[0]), cy: Math.round(entry.a[1]), rx: Math.round(entry.a[2]), ry: Math.round(entry.a[3]) });
        break;
      case 'fill':
        if (currentPath) {
          currentPath.fill = entry.fs || null;
          currentPath.lineWidth = entry.lw || 1;
          paths.push(currentPath);
          currentPath = null;
        }
        break;
      case 'stroke':
        if (currentPath) {
          currentPath.stroke = entry.ss || null;
          currentPath.lineWidth = entry.lw || 1;
          paths.push(currentPath);
          currentPath = null;
        }
        break;
    }
  }

  return {
    texts: texts,
    rects: rects,
    paths: paths.slice(0, 200),
    totalCalls: window.__canvasCallCount,
    logSize: log.length,
    capped: log.length >= 5000
  };
};
```

7. Add a re-render trigger function:
```javascript
window.triggerCanvasRerender = function() {
  // Clear log before re-render to get fresh data
  window.__canvasCallLog = [];
  window.__canvasCallCount = 0;

  // Strategy 1: resize event (most canvas apps re-render on resize)
  window.dispatchEvent(new Event('resize'));

  // Return a promise that resolves after two animation frames (render complete)
  return new Promise(resolve => {
    requestAnimationFrame(() => {
      requestAnimationFrame(() => {
        resolve({
          callCount: window.__canvasCallCount,
          logSize: window.__canvasCallLog.length
        });
      });
    });
  });
};
```

9. Do NOT use any Chrome extension APIs (no chrome.*, no browser.*). This runs in MAIN world.

  <action>
Add a `content_scripts` array to manifest.json. The entry MUST have these exact properties:

```json
"content_scripts": [
  {
    "matches": ["<all_urls>"],
    "js": ["canvas-interceptor.js"],
    "run_at": "document_start",
    "world": "MAIN"
  }
]
```

Place this after the `"web_accessible_resources"` array (at end of the JSON object, before the closing `}`).

Critical details:
- `"world": "MAIN"` -- this injects into the page's JS context, NOT the isolated content script world. This is required so the prototype wrapping affects the page's own canvas operations.
- `"run_at": "document_start"` -- injects before any page scripts run, ensuring we wrap prototypes before any canvas context is created.
- `"matches": ["<all_urls>"]` -- inject on every page because we do not know which pages will have canvas elements. The interceptor is a no-op on pages without canvas.

Do NOT modify any other fields in manifest.json. Preserve all existing content exactly as-is.
