**Case 3: mcp:run-agent**
```javascript
case 'mcp:run-agent': {
  const agentId = payload.agentId;
  if (!agentId) {
    sendMCPResponse(id, { success: false, error: 'Missing agentId' });
    break;
  }
  const agent = await agentManager.getAgent(agentId);
  if (!agent) {
    sendMCPResponse(id, { success: false, error: `Agent not found: ${agentId}` });
    break;
  }
  // Start execution - send initial ack, then run async
  // Store MCP message id for progress forwarding
  mcpProgressCallbacks.set(`agent:${agentId}`, id);
  try {
    const result = await agentExecutor.execute(agent);
    await agentManager.recordRun(agentId, {
      status: result.success ? 'success' : 'failed',
      result: result.summary || null,
      error: result.error || null,
      iterations: result.iterations || 0,
      tokensUsed: result.tokensUsed || 0,
      costUsd: result.costUsd || 0,
      durationMs: result.durationMs || 0,
      executionMode: result.executionMode || 'ai_initial',
      costSaved: result.costSaved || 0
    });
    sendMCPResponse(id, {
      success: result.success !== false,
      summary: result.summary || '',
      iterations: result.iterations || 0,
      durationMs: result.durationMs || 0,
      executionMode: result.executionMode || 'ai_initial'
    });
  } finally {
    mcpProgressCallbacks.delete(`agent:${agentId}`);
  }
  break;
}
```
