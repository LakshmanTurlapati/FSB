**2. Create `server/src/ws/handler.js`** -- the WebSocket room manager and message relay:

```javascript
const WebSocket = require('ws');

// Room map: hashKey -> { extensions: Set<ws>, dashboards: Set<ws> }
const rooms = new Map();

function setupWSHandler(wss) {
  wss.on('connection', (ws, request, { hashKey, role }) => {
    addClient(hashKey, ws, role);

    // Notify dashboards when extension connects
    if (role === 'extension') {
      broadcast(hashKey, 'dashboards', {
        type: 'ext:status', payload: { online: true }, ts: Date.now()
      });
    }

    ws.on('message', (data) => {
      try {
        const msg = JSON.parse(data);
        if (msg.type === 'ping') {
          ws.send(JSON.stringify({ type: 'pong', ts: Date.now() }));
          return; // Do NOT relay pings
        }
        // Relay to opposite side of room
        relayToRoom(hashKey, ws, data.toString());
      } catch { /* ignore malformed JSON */ }
    });

    ws.on('close', () => {
      removeClient(hashKey, ws);
      if (role === 'extension') {
        broadcast(hashKey, 'dashboards', {
          type: 'ext:status', payload: { online: false }, ts: Date.now()
        });
      }
    });

    ws.on('error', () => {}); // Prevent unhandled error crashes; onclose fires after
  });
}
```

Include helper functions: `addClient(hashKey, ws, role)`, `removeClient(hashKey, ws)`, `relayToRoom(hashKey, senderWs, rawMessage)`, `broadcast(hashKey, targetSide, messageObj)`, `broadcastToRoom(hashKey, messageObj)` (sends to ALL clients -- both extensions and dashboards -- used by agents route for REST-triggered events).

Export: `module.exports = { setupWSHandler, broadcastToRoom, rooms };`

The `broadcastToRoom` function is used by agents.js to push events when REST API modifies agents/runs. It sends to all connected dashboards in the room:
```javascript
function broadcastToRoom(hashKey, messageObj) {
  const room = rooms.get(hashKey);
  if (!room) return;
  const data = JSON.stringify(messageObj);
  for (const client of room.dashboards) {
    if (client.readyState === WebSocket.OPEN) {
      client.send(data);
    }
  }
}
```
