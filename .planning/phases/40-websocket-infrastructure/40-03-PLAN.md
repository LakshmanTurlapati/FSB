WebSocket endpoint: wss://host/ws?key={hashKey}&role=dashboard
- Server validates hashKey against DB on upgrade
- Server relays messages from extension to dashboard
- Server sends { type: 'ext:status', payload: { online: true|false }, ts: N } when extension connects/disconnects
- Server relays { type: 'ext:snapshot', payload: { version, timestamp }, ts: N } from extension on connect

REST API (unchanged, but now relative URLs):
- GET /api/auth/validate -- validate hashKey (header: X-FSB-Hash-Key)
- GET /api/agents -- list agents (header: X-FSB-Hash-Key)
- GET /api/stats -- get aggregate stats (header: X-FSB-Hash-Key)
- GET /api/agents/:id/runs?limit=N&offset=N -- paginated runs

WS message types the dashboard will receive:
- 'ext:status' -- { online: boolean } -- extension connected/disconnected
- 'ext:snapshot' -- extension state snapshot on connect
- 'agent_updated' -- agent CRUD event (from REST API broadcast)
- 'agent_deleted' -- agent deleted (from REST API broadcast)
- 'run_completed' -- run recorded (from REST API broadcast)
- 'pong' -- response to dashboard ping (ignore)

```javascript
function connectWS() {
  disconnectWS();
  var proto = location.protocol === 'https:' ? 'wss:' : 'ws:';
  var wsUrl = proto + '//' + location.host + '/ws?key=' +
    encodeURIComponent(hashKey) + '&role=dashboard';

  ws = new WebSocket(wsUrl);
  setWsState('reconnecting');

  ws.onopen = function () {
    wsReconnectDelay = 0;
    setWsState('connected');
  };

  ws.onmessage = function (event) {
    try {
      var msg = JSON.parse(event.data);
      handleWSMessage(msg);
    } catch (e) { /* ignore parse errors */ }
  };

  ws.onclose = function () {
    setWsState('disconnected');
    scheduleWSReconnect();
  };

  ws.onerror = function () {}; // onclose fires after
}
```

**4. Replace `disconnectSSE()` function (lines 436-442) with `disconnectWS()`:**

```javascript
function disconnectWS() {
  if (wsReconnectTimer) {
    clearTimeout(wsReconnectTimer);
    wsReconnectTimer = null;
  }
  if (ws) {
    ws.onclose = null; // Prevent reconnect on intentional close
    ws.close();
    ws = null;
  }
  setWsState('disconnected');
}
```

**5. Add `scheduleWSReconnect()` function:**

```javascript
function scheduleWSReconnect() {
  if (!hashKey) return;
  if (wsReconnectDelay === 0) {
    wsReconnectDelay = 1000;
    connectWS();
    return;
  }
  wsReconnectTimer = setTimeout(function () {
    connectWS();
  }, wsReconnectDelay);
  wsReconnectDelay = Math.min(wsReconnectDelay * 2, wsMaxReconnectDelay);
}
```

```javascript
function setWsState(state) {
  if (!sseStatusEl) return;
  var labels = {

    disconnected: 'disconnected',
    reconnecting: 'reconnecting...'

  sseStatusEl.textContent = labels[state] || state;

     state === 'reconnecting' ? 'dash-sse-reconnecting' :

}
```

```javascript
function handleWSMessage(msg) {
  if (msg.type === 'pong') return; // Ignore pong responses

  if (msg.type === 'ext:status') {
    extensionOnline = msg.payload && msg.payload.online;
    // Update agent count area to show extension status
    if (agentCountEl) {
      var countText = (stats.totalAgents || 0) + ' agent' +
        ((stats.totalAgents || 0) !== 1 ? 's' : '');
      agentCountEl.textContent = countText +
        (extensionOnline ? '' : ' - extension offline');
    }
    return;
  }

  if (msg.type === 'ext:snapshot') {
    extensionOnline = true;
    loadData(); // Refresh dashboard data on extension reconnect
    return;
  }

  // Agent/run events from REST API broadcasts
  if (msg.type === 'agent_updated' || msg.type === 'agent_deleted' || msg.type === 'run_completed') {
    loadData();
    if (msg.agentId && msg.agentId === selectedAgentId) {
      fetchRuns(selectedAgentId, runsLimit, runsOffset).then(function (result) {
        renderRuns(result.runs || [], result.total || 0, result.limit || runsLimit, result.offset || 0);
      }).catch(function () {});
    }
  }
}
```
