**JS changes in recreations.js:**

Add a new `initChartDraw()` function before the `init()` function (after `initStatusDots`). This function uses IntersectionObserver to add the `animate` class to `.rec-line-svg` when the chart scrolls into view, triggering the CSS stroke-dashoffset animation:

```javascript
// --- SVG line chart draw animation ---
function initChartDraw() {
  var svgs = document.querySelectorAll('.rec-line-svg');
  if (!svgs.length) return;

  svgs.forEach(function (svg) {
    var drawn = false;

    if ('IntersectionObserver' in window) {
      var observer = new IntersectionObserver(function (entries) {
        if (entries[0].isIntersecting && !drawn) {
          drawn = true;
          svg.classList.add('animate');
          observer.disconnect();
        }
      }, { threshold: 0.3 });
      observer.observe(svg);
    } else {
      svg.classList.add('animate');
    }
  });
}
```

Then add `initChartDraw();` to the `init()` function, right after `initChartBars();`.
