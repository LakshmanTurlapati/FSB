    ```typescript
    server.tool(
      "run_task",
      "Execute a browser automation task using FSB's AI. Describe what you want done in natural language and FSB's AI decides the steps. Returns a completion summary with success status and action log.",
      { task: z.string().describe("Natural language description of the task to perform") },
      async ({ task }) => {
        if (!bridge.isConnected()) {
          return mapFSBError({ success: false, error: 'extension_not_connected' });
        }
        return queue.enqueue("run_task", async () => {
          const result = await bridge.sendAndWait(
            { type: "mcp:start-automation", payload: { task } },

          );
          return mapFSBError(result.payload || result);
        });
      }
    );
    ```

    Use a data-driven approach -- define tool metadata in an array and loop to register. Each tool sends `mcp:execute-action` with `{ tool: fsbVerb, params: {...} }` to the extension.

    Helper function pattern for all manual tools:
    ```typescript
    async function execAction(bridge: NativeMessagingBridge, queue: TaskQueue, toolName: string, fsbVerb: string, params: Record<string, unknown>): Promise<ToolCallResult> {
      if (!bridge.isConnected()) {
        return mapFSBError({ success: false, error: 'extension_not_connected' });
      }
      return queue.enqueue(toolName, async () => {
        const result = await bridge.sendAndWait(
          { type: 'mcp:execute-action', payload: { tool: fsbVerb, params } },
          { timeout: 30_000 }
        );
        return mapFSBError(result.payload || result);
      });
    }
    ```
