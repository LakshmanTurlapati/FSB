/**
 * Get the relationship context of an element (form, navigation, region, modal)
 * @param {HTMLElement} element - The element to analyze
 * @returns {string|null} - Context string like "in login form" or "in main navigation"
 */
function getRelationshipContext(element) {
  // Check for modal/dialog context (highest priority - overrides other context)
  const modal = element.closest('[role="dialog"], [role="alertdialog"], [class*="modal"], [class*="dialog"], [class*="popup"]');
  if (modal) {
    const modalLabel = modal.getAttribute('aria-label') || modal.getAttribute('aria-labelledby');
    if (modalLabel) {
      const labelText = document.getElementById(modalLabel)?.textContent?.trim() || modalLabel;
      return `in "${labelText.substring(0, 30)}" modal`;
    }
    return 'in modal dialog';
  }

  // Check for form context
  const form = element.closest('form');
  if (form) {
    // Try to identify the form's purpose
    const formId = form.id || form.name;
    const formAction = (form.action || '').toLowerCase();
    const formAriaLabel = form.getAttribute('aria-label');

    // Infer form type from action URL or content
    if (formAriaLabel) {
      return `in "${formAriaLabel.substring(0, 30)}" form`;
    }
    if (/login|signin|sign-in/.test(formAction) || form.querySelector('input[type="password"]')) {
      return 'in login form';
    }
    if (/signup|register|sign-up|create/.test(formAction)) {
      return 'in signup form';
    }
    if (/search/.test(formAction) || form.querySelector('[type="search"]')) {
      return 'in search form';
    }
    if (/checkout|payment|order/.test(formAction)) {
      return 'in checkout form';
    }
    if (/contact/.test(formAction)) {
      return 'in contact form';
    }
    if (/subscribe|newsletter/.test(formAction)) {
      return 'in newsletter form';
    }
    if (formId) {
      return `in "${formId}" form`;
    }
    return 'in form';
  }

  // Check for navigation context
  const nav = element.closest('nav, [role="navigation"]');
  if (nav) {
    const navLabel = nav.getAttribute('aria-label');
    if (navLabel) {
      return `in "${navLabel.substring(0, 30)}" navigation`;
    }
    // Try to determine nav type from position
    const rect = nav.getBoundingClientRect();
    if (rect.top < 100) {
      return 'in header navigation';
    }
    if (rect.top > window.innerHeight - 200) {
      return 'in footer navigation';
    }
    // Check for sidebar
    if (rect.width < window.innerWidth * 0.3 && (rect.left < 50 || rect.right > window.innerWidth - 50)) {
      return 'in sidebar navigation';
    }
    return 'in navigation';
  }

  // Check for header/footer context
  const header = element.closest('header, [role="banner"]');
  if (header) {
    return 'in header';
  }

  const footer = element.closest('footer, [role="contentinfo"]');
  if (footer) {
    return 'in footer';
  }

  // Check for aside/sidebar context
  const aside = element.closest('aside, [role="complementary"]');
  if (aside) {
    return 'in sidebar';
  }

  // Check for main content context
  const main = element.closest('main, [role="main"]');
  if (main) {
    return 'in main content';
  }

  // Check for article context
  const article = element.closest('article');
  if (article) {
    return 'in article';
  }

  // Check for list context (useful for repeated items)
  const listItem = element.closest('li');
  if (listItem) {
    const list = listItem.closest('ul, ol, [role="list"]');
    if (list) {
      const listLabel = list.getAttribute('aria-label');
      if (listLabel) {
        return `in "${listLabel.substring(0, 20)}" list`;
      }
    }
  }

  // Check for card/item context (common in modern UIs)
  const card = element.closest('[class*="card"], [class*="item"], [class*="tile"]');
  if (card) {
    // Try to get card title
    const cardTitle = card.querySelector('h1, h2, h3, h4, h5, h6, [class*="title"]');
    if (cardTitle) {
      const title = cardTitle.textContent?.trim().substring(0, 25);
      if (title) {
        return `in "${title}" card`;
      }
    }
    return 'in card';
  }

  return null;
}
