```javascript
#!/usr/bin/env node
// Phase 216 SMOKE-01..03: production-or-local crawler smoke.
// Curls marketing routes under GPTBot UA; curls crawler files; resolves every sitemap <loc>.
// Run: BASE_URL=http://localhost:3000 node scripts/smoke-crawler.mjs   (local)
//      node scripts/smoke-crawler.mjs                                  (production default)
// Zero new npm deps -- node:fetch (built-in Node 18+), no other imports needed for assertions.

const BASE_URL = (process.env.BASE_URL || 'https://full-selfbrowsing.com').replace(/\/$/, '');
const PROD_HOST = 'https://full-selfbrowsing.com';
const UA = 'GPTBot';

const failures = [];
const passes = [];

function record(passed, label, detail) {
  (passed ? passes : failures).push({ label, detail });
  console.log(`${passed ? 'PASS' : 'FAIL'}: ${label}${detail ? ` -- ${detail}` : ''}`);
}

async function fetchText(url, opts = {}) {
  const res = await fetch(url, { redirect: 'follow', headers: { 'User-Agent': opts.ua || UA }, ...opts });
  const body = await res.text();
  return { status: res.status, contentType: res.headers.get('content-type') || '', body };
}

const MARKETING_ASSERTIONS = [
  { path: '/',        titleSubstr: 'Full Self-Browsing', canonical: `${PROD_HOST}` },
  { path: '/about',   titleSubstr: 'About',              canonical: `${PROD_HOST}/about` },
  { path: '/privacy', titleSubstr: 'Privacy',            canonical: `${PROD_HOST}/privacy` },
  { path: '/support', titleSubstr: 'Support',            canonical: `${PROD_HOST}/support` },
];

async function checkMarketingRoutes() {
  for (const { path, titleSubstr, canonical } of MARKETING_ASSERTIONS) {
    const url = `${BASE_URL}${path}`;
    let r;
    try { r = await fetchText(url); } catch (err) { record(false, `GET ${path}`, `fetch threw: ${err.message}`); continue; }
    record(r.status === 200, `GET ${path} -> 200`, `actual ${r.status}`);
    record(/text\/html/i.test(r.contentType), `GET ${path} content-type is text/html`, r.contentType);
    record(r.body.includes(`>FSB`) && r.body.includes(titleSubstr), `GET ${path} title contains "${titleSubstr}"`, '');
    record(r.body.includes(`href="${canonical}"`), `GET ${path} canonical href="${canonical}"`, '');
    record(r.body.includes('<app-root'), `GET ${path} contains <app-root>`, '');
    if (path === '/') {
      record(r.body.includes('<script type="application/ld+json"') || r.body.includes('type="application/ld+json"'), `GET / contains JSON-LD <script>`, '');
    }
  }
}

const CRAWLER_FILES = [
  { path: '/robots.txt',    ctMatch: /text\/plain/i,       bodyMustContain: 'User-agent: GPTBot' },
  { path: '/sitemap.xml',   ctMatch: /xml/i,               bodyMustContain: '<urlset' },
  { path: '/llms.txt',      ctMatch: /text\/plain/i,       bodyMustContain: '# FSB (Full Self-Browsing)' },
  { path: '/llms-full.txt', ctMatch: /text\/plain/i,       bodyMustContain: '' },
];

async function checkCrawlerFiles() {
  for (const { path, ctMatch, bodyMustContain } of CRAWLER_FILES) {
    const url = `${BASE_URL}${path}`;
    let r;
    try { r = await fetchText(url, { ua: 'curl/7.0' }); } catch (err) { record(false, `GET ${path}`, `fetch threw: ${err.message}`); continue; }
    record(r.status === 200, `GET ${path} -> 200`, `actual ${r.status}`);
    record(ctMatch.test(r.contentType), `GET ${path} content-type matches ${ctMatch}`, r.contentType);
    record(r.body.length > 0, `GET ${path} body non-empty`, `${r.body.length} bytes`);
    if (bodyMustContain) {
      record(r.body.includes(bodyMustContain), `GET ${path} body contains "${bodyMustContain}"`, '');
    }
    if (path === '/llms-full.txt') {
      record(Buffer.byteLength(r.body, 'utf8') < 256000, `GET ${path} body < 256000 bytes`, `${Buffer.byteLength(r.body, 'utf8')}`);
    }
  }
}

async function checkSitemapLocs() {
  const url = `${BASE_URL}/sitemap.xml`;
  let r;
  try { r = await fetchText(url, { ua: 'curl/7.0' }); } catch (err) { record(false, `parse sitemap`, err.message); return; }
  if (r.status !== 200) { record(false, `parse sitemap`, `sitemap status ${r.status}`); return; }
  const locs = [...r.body.matchAll(/<loc>([^<]+)<\/loc>/g)].map((m) => m[1].trim());
  record(locs.length === 4, `sitemap has 4 <loc> entries`, `actual ${locs.length}`);
  for (const loc of locs) {
    // Translate prod-host references to BASE_URL when running locally so the assertion exercises the local server.
    const target = BASE_URL === PROD_HOST ? loc : loc.replace(PROD_HOST, BASE_URL);
    let lr;
    try { lr = await fetchText(target); } catch (err) { record(false, `sitemap loc ${loc}`, `fetch threw: ${err.message}`); continue; }
    record(lr.status === 200, `sitemap loc ${loc} -> 200`, `actual ${lr.status}`);
    record(lr.body.includes('<app-root'), `sitemap loc ${loc} body contains <app-root>`, '');
  }
}

async function main() {
  console.log(`[smoke-crawler] BASE_URL=${BASE_URL}`);
  await checkMarketingRoutes();
  await checkCrawlerFiles();
  await checkSitemapLocs();
  console.log('');
  console.log(`[smoke-crawler] passed=${passes.length} failed=${failures.length}`);
  if (failures.length > 0) {
    console.log('[smoke-crawler] FAILURES:');
    for (const f of failures) console.log(`  - ${f.label}${f.detail ? ` (${f.detail})` : ''}`);
    process.exit(1);
  }
  console.log('[smoke-crawler] all assertions passed');
}

main().catch((err) => {
  console.error('[smoke-crawler] FATAL:', err.message);
  process.exit(2);
});
```

Notes:
- Per CONTEXT.md "Claude's Discretion", report format is plain text. Acceptable; the constraint is just exit-code semantics (0 = pass, non-zero = at least one failure).
- The script does NOT use node:assert in the throw-on-fail sense -- it collects failures and reports all. CONTEXT.md D-12 references node:assert as a primitive but the operational requirement is the exit-code contract; collect-and-report is a better DX for a smoke script.
- node:fetch is built-in Node 18+. The repo already builds with Node 18+ (Angular 19 requires it). No `npm install` step needed.
- No emojis in any console output per project CLAUDE.md.
- The local-mode loc-rewrite (`loc.replace(PROD_HOST, BASE_URL)`) handles the case where the local server's sitemap still hardcodes `https://full-selfbrowsing.com` URLs (which it does -- the URLs are content, not requests). Without this, local mode would attempt to hit production for the loc assertions.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| BASE_URL env var -> fetch target | The smoke script trusts the operator's BASE_URL choice. There is no auth or network secret involved; misuse is bounded to "fetch the wrong site". |
| Live HTTP responses -> assertion logic | The script accepts whatever the target returns and asserts content. A compromised target could lie about its content, but the script's purpose is to detect drift, not to defend against active man-in-the-middle. HTTPS in production mode is the load-bearing protection. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-216-T4 | Tampering | smoke script silently passes when target is wrong | mitigate | Script asserts route-specific title substrings (`About`, `Privacy`, `Support`, `Full Self-Browsing`) AND route-specific canonical hrefs AND `<app-root>` presence -- a target serving the wrong page or no prerender at all fails multiple assertions, not just one. Verify-smoke.sh acceptance includes "breaking the script causes non-zero exit" to prove the assertions are real. |
| T-216-S2 | Spoofing | local-mode running against production by accident | accept | BASE_URL is operator-controlled; the verify-smoke.sh script always sets BASE_URL=http://localhost:<port> explicitly, so accidental production hits are bounded to manual `npm run smoke:crawler` invocations -- which is the intended production smoke path anyway. |
| T-216-D2 | Denial of service | smoke script hangs on a slow target | accept | node:fetch's default behavior is to wait until the target responds; a slow target produces a slow smoke run. CI timeouts catch this at the orchestration layer. Adding explicit timeouts is a Future enhancement; not required for v0.9.46. |

T-216-01 (crawler file shadowing) is enforced at the server level by Plan 03; this plan's smoke is an additional end-to-end check (it asserts Content-Type matches `text/plain` for /robots.txt etc., which would fail if the SPA fallback shadowed the file).

T-LD-01 (Phase 215 carry-forward, JSON-LD payload integrity): the smoke asserts `<script type="application/ld+json">` presence on home but does not parse the payload. Full payload validation is the Rich Results Test, owned by Plan 05's manual UAT.
</threat_model>
