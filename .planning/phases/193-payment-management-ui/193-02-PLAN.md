   ```javascript
   function detectPaymentCardBrand(cardNumber) {
     const digits = String(cardNumber || '').replace(/\D/g, '');
     if (!digits) return 'unknown';
     if (/^4/.test(digits)) return 'visa';
     if (/^(5[1-5]|2[2-7])/.test(digits)) return 'mastercard';
     if (/^3[47]/.test(digits)) return 'amex';
     if (/^(6011|65|64[4-9])/.test(digits)) return 'discover';
     return 'unknown';
   }

   function isValidPaymentCardNumber(cardNumber) {
     const digits = String(cardNumber || '').replace(/\D/g, '');
     if (!/^\d{12,19}$/.test(digits)) return false;
     let sum = 0, shouldDouble = false;
     for (let i = digits.length - 1; i >= 0; i--) {
       let digit = Number(digits[i]);
       if (shouldDouble) { digit *= 2; if (digit > 9) digit -= 9; }
       sum += digit;
       shouldDouble = !shouldDouble;
     }
     return sum % 10 === 0;
   }

   function isValidPaymentExpiry(month, year) {
     const m = parseInt(month, 10);
     const y = year.length === 2 ? 2000 + parseInt(year, 10) : parseInt(year, 10);
     if (!m || m < 1 || m > 12 || !y) return false;
     const now = new Date();
     if (y < now.getFullYear()) return false;
     if (y === now.getFullYear() && m < now.getMonth() + 1) return false;
     return true;
   }
   ```

| options.js modal -> background.js savePaymentMethod | Full card data (number, CVV) crosses message boundary during save |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-193-04 | Information Disclosure | showPaymentMethodModal edit mode | accept | Full card data shown in edit modal is acceptable -- user already proved access via PIN unlock; auto-hide CVV after modal close by clearing fields in hidePaymentMethodModal |
| T-193-05 | Tampering | savePaymentMethodFromModal | mitigate | Client-side Luhn validation prevents obviously invalid card numbers from being saved; backend also validates via isValidCardNumber |
| T-193-06 | Denial of Service | deletePaymentMethodConfirm | mitigate | confirm() dialog prevents accidental deletion; single-action (no batch delete) limits blast radius |
| T-193-07 | Tampering | innerHTML in renderPaymentCard | mitigate | All user data passes through escapeHtml before insertion (from Plan 01) |
</threat_model>
