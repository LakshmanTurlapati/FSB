```javascript
function getMatchedPlatforms(flags) {
  const matched = [];
  for (const [key, platform] of Object.entries(PLATFORMS)) {
    if (flags[key] === true && platform.format !== 'cli') {
      matched.push({ key, platform });
    }
  }

}
```

**3. Add `handleClaudeCodeInstall()` function** (place it after `printUninstallUsage()`, before `runInstall()`):

```javascript
function handleClaudeCodeInstall() {
  const cmd = 'claude mcp add --scope user ' + FSB_SERVER_NAME + ' -- npx -y fsb-mcp-server';
  try {
    execSync(cmd, { stdio: 'pipe', encoding: 'utf-8' });
    console.log('\u2713 Added to Claude Code');
    return { installed: true, skipped: false };
  } catch (err) {
    if (err.stdout?.includes('already exists') || err.stderr?.includes('already exists')) {
      console.log('\u25CB Claude Code already configured');
      return { installed: false, skipped: true };
    }
    console.log('\u25CB Claude CLI not found \u2014 run manually: claude mcp add ' + FSB_SERVER_NAME + ' -- npx -y fsb-mcp-server');
    return { installed: false, skipped: false };
  }
}
```
Uses `--scope user` per research recommendation A1 (FSB is user-global, matching all other platform installs). Per D-01, shells out to `claude mcp add` via `execSync`, fallback prints the command if CLI not found. Handles idempotency by catching "already exists" exit code 1 (per Pitfall 1 in research).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI flags -> platform selection | User-provided flags matched against hardcoded PLATFORMS registry keys |
| install.js -> child_process.execSync | Shells out to `claude` CLI binary for Claude Code |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-176-01 | Tampering | handleClaudeCodeInstall execSync | mitigate | Command string constructed entirely from hardcoded constants (FSB_SERVER_NAME = 'fsb'). No user input interpolated into the shell command. |
| T-176-02 | Spoofing | claude binary in PATH | accept | User controls their own PATH. If a malicious `claude` binary exists, this is a broader system compromise beyond FSB's scope. |
| T-176-03 | Information Disclosure | execSync error output | mitigate | Using `stdio: 'pipe'` to suppress raw CLI output. Error messages are generated from hardcoded strings, not from raw error objects. |
</threat_model>
