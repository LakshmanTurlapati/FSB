```javascript
window.fetch = function(...args) {
  pendingRequestCount++;
  lastNetworkActivity = Date.now();
  return originalFetch.apply(this, args).finally(() => {
    pendingRequestCount--;
    lastNetworkActivity = Date.now();
  });
};
```
